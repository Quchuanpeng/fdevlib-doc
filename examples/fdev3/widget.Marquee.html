fdevlib-example: marquee（走马灯效果）

1、自动向左滚动、每次滚1格

调用代码示意：

						var mar1 = new FD.widget.Marquee('mar1');

2、自动(需启动)向上滚动、每次滚1格，存在暂停和启动按钮

调用代码示意：

						var mar2 = new FD.widget.Marquee('mar2',{
							direction: 'up',
							isAutoPlay: false
						});
						$E.on('pause','click',function(){
							mar2.pause();
						});
						$E.on('start','click',function(){
							mar2.start();
						});

3、自动向右滚动、每次滚3格

调用代码示意：

					var mar3 = new FD.widget.Marquee('mar3',{
						preItem: 3,
						direction:'right'
					});

4、匀速左滚、间隔5秒、每次滚动耗时0.6秒、有前/后按钮

调用代码示意：

					var mar4 = new FD.widget.Marquee('mar4',{
						preItem: 3,
						timeDelay: 5,
						easing: 'easeNone',
						speed: 0.6
					});
					$E.on('fw-pre','click',function(){
						mar4.getPre();
					});
					$E.on('fw-next','click',function(){
						mar4.getNext();
					});

5、不循环fill模式

调用代码示意：

					var mar5 = new FD.widget.Marquee('mar5',{
						isAutoPlay: false,
						direction: 'left',
						preItem: 3,
						loopType: 'fill',
						onNextEnd: function(){
							FD.common.toggleClass('fw-next3','next','next-g');
						},
						onPreEnd: function(){
							FD.common.toggleClass('fw-pre3','pre','pre-g');
						},
						onNextRs :function(){
							FD.common.toggleClass('fw-next3','next','next-g');
						},
						onPreRs: function(){
							FD.common.toggleClass('fw-pre3','pre','pre-g');
						}
					});	
					$E.on('fw-pre3','click',function(){
						mar5.getPre();
					});
					$E.on('fw-next3','click',function(){
						mar5.getNext();
					});
