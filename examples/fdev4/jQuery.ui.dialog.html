ui-dialog : 浮出层

多重浮出层

Drag me!

本浮出层将在打开后1秒后关闭！ 

<!--HTML-->            	
<div class="dialog">
    <div class="header">Drag me!</div>
    <p>
        <button class="close">关闭</button>
        <button id="trigger3">打开</button>
    </p>
</div>
<div class="dialog">
    <p>本浮出层将在打开后1秒后关闭！</p>
</div>
			
/*Javascript*/ 	
var dialogs = $('div.dialog', '#demo1');
$('#trigger1').click(function(){
    $(dialogs[0]).bind('dialogbefore', function(){
        $.log('custom before');
    }).dialog({
        modalCss: {
            backgroundColor: '#FFF'
        },
        draggable: {
            handle: 'div.header'
        },
        shim: true,
        open: function(){
            $.log('open');
        },
        beforeClose: function(){
            $.log('beforeClose');
            return true;
        },
        close: function(){
            $.log('close');
        }
    });
});	
//注册多重浮出层
$('#trigger3').click(function(){
    $(dialogs[1]).dialog({
        fadeIn: 500,
        fadeOut: 500,
        timeout: 1000,
        center: true
    });
});
//注册弹出框内关闭事件
$('button.close', '#demo1').click(function(){
    $(this).closest('div.dialog').dialog('close');
});

无蒙版浮出层

Drag me! 

<!--HTML-->             	
<div class="dialog">
    <p>
        Drag me! <button class="close">关闭</button>
    </p>
</div>
			
/*Javascript*/         	
var dialog2 = $('div.dialog', '#demo2');			
$('#trigger2').click(function(event){
    dialog2.dialog({
        modal: false,
        shim: true,
        draggable: true,
        css: {
            left: event.clientX + 50,
            top: event.clientY - 20
        }
    });
});
