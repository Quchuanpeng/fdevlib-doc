ui-progressbar : 进度条

Demo

<div id="progressbar"></div>
.ui-progressbar{
    width: 200px;
    height: 20px;
    border: 1px solid #ddd;
    color: #333;
    border-radius: 4px;
}
.ui-progressbar-value{
    border: 1px solid #E78F08;
    background: #F6A828;
    color: white;
    font-weight: bold;
    height:100%;
    margin: -1px;
    border-radius: 4px;
}
jQuery(function($){
    $.use("ui-progressbar",function(){
        $("#progressbar").progressbar({
            value: 50,
            complete: function(event, ui){
                alert("complete!");
            }
        });
                  
        $("#progressbar-value").bind("change click",function(){
            value = $("#progressbar-value").val();
            // 更改value，进度条将会自动刷新
            $("#progressbar").progressbar("value",parseInt(value));
        });
    });
});
