jQuery(function($){
    $.use('ui-tabs-effect', function(){
        var tab6 = $('#tab6').tabs({
            isAutoPlay:false,
            boxSelector:'li',
            effect:'scroll',
            scrollType:'loop',
            perItem:3
        });
        //绑定左右滚动事件
        $('#fw-pre6').bind('click', function(){
            tab6.tabs('prev');
        });
        $('#fw-next6').bind('click', function(){
            tab6.tabs('next');
        });
    });
});

6、tabs切换 - 滚动效果

jQuery(function($){
    $.use('ui-tabs-effect', function(){
        var tabs7 = $('#tabs7').tabs({
            isAutoPlay:false,
            boxSelector:'li',
            effect:'scroll',
            perItem:3
        });
        var tabs7L = tabs7.tabs('length');
        tabs7.bind('tabsshow', function(e, data){
            $('#fw-next7').toggleClass('next-g', data.index===tabs7L-1);
            $('#fw-next7').toggleClass('next', data.index!==tabs7L-1);
            
            $('#fw-pre7').toggleClass('pre-g', data.index===0);
            $('#fw-pre7').toggleClass('pre', data.index!==0);
        });
        //绑定左右滚动事件
        $('.mar7 .pre').live('click', function(){
            tabs7.tabs('prev');
        });
        $('.mar7 .next').live('click', function(){
            tabs7.tabs('next');
        });
    });
});
