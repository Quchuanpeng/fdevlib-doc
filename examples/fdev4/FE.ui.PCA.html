<!--HTML-->
<div id="demo">
    <select></select>
    <select></select>

// JavaScript
jQuery(function($){
    // 动态载入PCA数据
    $.add("web-pca-data",{
        requires: ["web-pca"],
        js: ["http://" + $.styleDomain + "/js/lib/fdev-v4/widget/web/pca-data-min.js"]
    });
    $.use("web-pca-data",function(){
        new FE.ui.PCA($("select","#demo"),{
            store: FE.ui.PCA.data,
            onPSelected: function(){    
                // 第一个下拉框选项变动后触发，this指向下拉框本身 
            },
            onCSelected: function(){
                // 第二个下拉框选项变动后触发，this指向下拉框本身 
            },
            onASelected: function(){
                // 第三个下拉框选项变动后触发，this指向下拉框本身 
            }
        });
    });
});
