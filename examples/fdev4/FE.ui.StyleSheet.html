<!--HTML-->
<div id="demo1">
    <span>span</span>
    <em>em</em>
    <a href="#">link</a>
    <div>
        <button id="enable">enable</button>
        <button id="disable">disable</button>
    </div>
</div>
/* JavaScript */
jQuery(function($){
    $.use("web-stylesheet",function(){
        var demo1 = new FE.ui.StyleSheet();
        demo1.set("#demo1 span",{
            color: 'red'
        });
        demo1.set("#demo1 em",{
            color: 'blue'
        });
        demo1.set("#demo1 a:hover",{
            color: '#000'
        });

        $("#enable").click(function(){
            demo1.enable();
        });
        $("#disable").click(function(){
            demo1.disable();
        });
        
        console.log(demo1.cssRules);
    });
});
