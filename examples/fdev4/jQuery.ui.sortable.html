jQuery(function($){
    $.use("ui-sortable",function(){
        $("#demo2-sortable1,#demo2-sortable2").sortable({
            connectWith: ".demo2",      // 就是靠这个属性将两个list联系起来的
            placeholder: "placeholder"  // 给的是类名,名字前面不用带"."
        });
    });
});

结果也可以在控制台查看

<div id="demo3">
    <ul id="demo3-sortable">
        <li id="demo3-item-1">item 1</li>
        <li id="demo3-item-2">item 2</li>
        <li id="demo3-item-3">item 3</li>
        <li id="demo3-item-4">item 4</li>
    </ul>
    <button id="serialize">获取序列化后的值</button>
    <button id="toArray">To Array</button>
</div>
jQuery(function($){
    $.use("ui-sortable",function(){
        $("#demo3-sortable").sortable(); 
                    
        $("#serialize").click(function(){
            alert($("#demo3-sortable").sortable("serialize"));
            console.log($("#demo3-sortable").sortable("serialize"));
        });
                    
        $("#toArray").click(function(){
            alert($("#demo3-sortable").sortable("toArray"));
            console.log($("#demo3-sortable").sortable("toArray"));
        });
    });
});
