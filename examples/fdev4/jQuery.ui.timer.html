ui-timer : 计时器

Demo 1 文字模式

年 月 天 小时 分钟 秒 

<!--HTML-->
<p id="timer1">
    <span class="year"></span>年
    <span class="month"></span>月
    <span class="day"></span>天
    <span class="hour"></span>小时
    <span class="minute"></span>分钟
    <span class="second"></span>秒
</p>
/* Javascript */
jQuery(function($){
    $.use("ui-timer",function(){
        $("#timer1").timer({
            to: new Date(2020, 7, 5, 12, 12, 12)
        });
    });
});

Demo 2 图片模式

年
月
天
小时
分钟
秒
<!--HTML-->
<div id="timer2">
    <div class="year"></div><div>年</div>
    <div class="month"></div><div>月</div>
    <div class="day"></div><div>天</div>
    <div class="hour"></div><div>小时</div>
    <div class="minute"></div><div>分钟</div>
    <div class="second"></div><div>秒</div>
</div>
/* Javascript */
jQuery(function($){
    // 加载图片模式所需要的图片样式
    $.loadCSS('http://'+ $.styleDomain + '/css/lib/fdev-v4/widget/ui/timer-min.css');
    $.use("ui-timer",function(){
        $("#timer2").timer({
            animate: true,
            to: new Date(2020, 7, 5, 12, 12, 12)
        });
    });
});
