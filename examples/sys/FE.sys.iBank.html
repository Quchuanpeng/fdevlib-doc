//在没有Figo配置生效的环境下，手动配置Figo项
FE.test['style.picman.url'] = 'http://picman-test.china.alibaba.com:40100';
jQuery(function($){
    //配置图片银行模块
    $.add('sys-ibank', {
        css: ['http://style.china.alibaba.com/css/sys/ibank/ibank-min.css'],
        js: ['http://style.china.alibaba.com/js/sys/ibank/ibank-min.js'],
        ver: '1.0'
    });
    //静态页面获取token
    var tokenRequest = $.ajax('http://picman-test.china.alibaba.com:40100/open/ajax/csrf_token_get.json', {
        dataType: 'jsonp'
    });
    
    $('button', '#demo1').click(function(e){
        $.use('sys-ibank', function(){
            $.when(tokenRequest).then(function(o){
                FE.sys.iBank({
                    tabs: ['album', 'upload'],
                    albumAuth: false,
                    watermark: false,
                    submitText: '确定',
                    //portlets: false,
                    //albumId: '15815320',
                    //allowMultiple: false,
                    dialog: {
                        open: function(){
                            //alert('组件打开');
                        }
                    },
                    //tab: 'upload',
                    token: o._csrf_token,
                    source: 'picman',
                    insert: function(selected){
                        $.log(selected);
                    }
                });
                
            });
        });
    });
});
