CSS/JS依赖：

/* CSS */
@import url("http://style.china.alibaba.com/css/sys/popsignin/loginreg.css");
/* 此CSS依赖于任何版本的reset.css及fdev-v3的type.css，优先使用 */
            
/* JS */
ImportJavscript.url("http://style.china.alibaba.com/js/sys/popsignin/loginreg.js");
/* 此JS依赖于fdev.js、get.js，fdevlib/fdev-v3版本均可，但fdev、get须同版本 */

使用方法：

/*
 * @param {Object} options:
 *	showOverlay {Bolean} 是否显示遮罩
 *	onShow {Function} 浮出后触发回调函数
 *	onLoginSuccess {Function} 登陆成功之后回调函数
 *		@this	成功后的div容器
 *		@param	提交的get参数
 *		@o		后台返回的数据
 *	onRegistSuccess {Function} onRegistSuccess 注册成功之后回调函数
 *		@this	成功后的div容器
 *		@param	提交的get参数
 *		@o		后台返回的数据
 *	imgDomain {String} 验证图片的前缀，默认为http://checkcode.china.alibaba.com
 *	aliDomain {String} 数据请求的前缀，默认为http://china.alibaba.com
 */		
FD.Member.LR.show(options);
FD.Member.LR.hide();
