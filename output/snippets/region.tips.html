//Mouse Click
(function() {
    var link1=FYG('link1'),
        tip1=FYG('tip1');
    FYE.on(link1,'click',function(e) {
        FYE.preventDefault(e);
        if(FYD.getStyle(tip1,'display')!='none') return;
        var rlink=FYD.getRegion(this);
        FYD.setStyle(tip1,'left',rlink.left-80+'px');
        FYD.setStyle(tip1,'top',rlink.top+22+'px');
        FYD.setStyle(tip1,'display','');
        function rHandler(e) {
            var target=FYE.getTarget(e);
            if(link1!=target&&tip1!=target&&!FYD.isAncestor(tip1,target)) {
                FYD.setStyle(tip1,'display','none');
                FYE.removeListener(document,'click',rHandler);
            }
        }
        FYE.on(document,'click',rHandler);
    });
})();
//Mouse Leave
(function() {
    var link2=FYG('link2'),
        tip2=FYG('tip2'),
        timer;

    function clear() {
        if(timer) {
            clearTimeout(timer);
            timer=null;
        }
    }
    function hide() {
        clear();
        timer=setTimeout(function() {
            FYD.setStyle(tip2,'display','none');
        },500);
    }
    FYE.on(link2,'mouseover',function(e) {
        clear();
        if(FYD.getStyle(tip2,'display')!='none') return;
        var rlink=FYD.getRegion(this);
        FYD.setStyle(tip2,'left',rlink.left-80+'px');
        FYD.setStyle(tip2,'top',rlink.top+22+'px');
        FYD.setStyle(tip2,'display','');
    });
    FYE.on(link2,'mouseout',hide);
    FYE.on(tip2,'mouseenter',clear);
    FYE.on(tip2,'mouseleave',hide);
})();
