正则

Regex：常用正则表达式（部分由俞雨友情赞助）

js: 
        //匹配前后空格，包括全角空格
        new RegExp('^[\s\u00A0\u3000]+|[\s\u00A0\u3000]+$','g');
        //中文站统一E-mail地址
        new RegExp('^[\w\-]+(\.[\w\-]*)*@[\w\-]+([\.][\w\-]+)+$');      
        //匹配中文
        new RegExp('^[\u0391-\uFFE5]+$');      
        //数值，包括正负及浮点
        new RegExp('^[-+]?\d+(\.\d+)?$');      
        //手机号码
        new RegExp('^[1][3|5|8]\d{9}$');      
        //邮政编码
        new RegExp('^\d{6}$');
