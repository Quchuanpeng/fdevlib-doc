grid.js : 列表组件

介绍

grid组件迁移自xui的simple-grid组件，基于jQuery和后台宏函数实现，提供了列表的方法及属性。

版本

  • 盛鼎：初始版1.0 ---- 2011-09-26

开始

<--引入对应的主题样式文件。注意：目前在grid组件目录下包含多套主题，请使用的同学根据不同应用进行匹配，若新的应用需要使用本组件，请另行创建新的样式文件-->
<link href="http://style.china.alibaba.com/css/lib/fdev-v4/widget/web/grid/tbcall/simple-grid/simple-grid.css" rel="stylesheet"/>
<--引入组件功能JS文件-->
<script src="http://style.china.alibaba.com/js/lib/fdev-v4/widget/web/grid.js"></script>                
            
 #SimpleGrid("grid-example" ,[["width","675px"],["limit",10],["hasPagebar",true],["checkbox",true],["colModel",[["sex","100px","性别"],["age","100px","年龄"],["birthday","100px","生日"],["name","100px","姓名"],["opt","100px","操作"]]]])

var grid = new FE.ui.Grid('grid-example',options);
grid.init(options);

Configs

limit : Number
Default : 10
分页显示的数目
toPageNum : Number
Default : 1
初始化之后显示第几页数据，默认是第一页
checkbox : Boolean
Default : false
是否有checkbox选择项
hasPages : Boolean
Default : false
是否有总页数显示
hasPagebar : Boolean
Default : true
是否有翻页条
isRowAuto : Boolean
Default : true
是否根据数据的数目显示多少行
defaultSort : String
Default : ''
默认排序字段设置
dir : String
Default : 'desc'
排序顺序 , 'desc'或者‘asc’
multiSelect : Boolean
Default : false
是否可以多选

  • obj :Object 单行数据json对象，如{"nanme":"jack","age":19}

  • e :Event 事件对象
  • hdId :String 排序的列id

Methods

init(sUrl,oColModel,oSchema)
  • sUrl : String 请求数据地址的url
  • oColModel : Object 列模型对象实例
  • oSchema : Object 数据模型
初始化Grid。
    var dataUrl = 'http://style.china.alibaba.com/data.html';
    var schema =['sex','age','birthday','name','status'];
    
    var colModel = new FE.ui.GridColModel({
                defaults: { 
                    width: '10%',
                    sortable: true 
                },
                columns: [
                    {id: 'sex', header: '性别', dataIndex: 'sex'} ,
                    {id: 'age', header: '年龄', dataIndex: 'age'} ,
                    {id: 'birthday', header: '日期', dataIndex: 'birthday',align:'left'},
                    {id: 'name', header: '名字', dataIndex: 'name',align:'right'},
                    {id: 'opt', header: '操作',sortable: false }  
                ]
            })
    var g = new FE.ui.Grid('grid-example'); 
    g.init(dataUrl,colModel,schema)
sort(hdId,dir,beFirst)
  • hdId : String 列的ID
  • dir : String 排序方式 'desc'和'asc',
  • beFirst : Boolean 排序是否回到第一页，默认为false
给Grid排序。
    g.sort('sex','asc',true);
load()
刷新当前页数据。
    g.load();
reload(oParams)
  • oParams : Object 如 {"searchKeyWord":"test"}
设置获取数据的参数,并返回第一页。
    var oparms = {"searchKeyWord":"test"};
    g.reload(oparms);
selectAll()
选中所有行。
    g.selectAll();
selectRowByIndex(rowIndex)
  • rowIndex : Number 根据制定的序号选中行
根据指定的序号选中行。
    g.selectRowByIndex(10);
getSelectRowIndex()
  • return : Number 行的序号
返回当前选中行的序号。
    g.getSelectRowIndex();
getMouseoverRowIndex()
  • return : Number 行的序号
返回当前鼠标移动上行的序号。
    g.getMouseoverRowIndex();
unselectRowByIndex(rowIndex)
  • rowIndex : Number 行的序号
取消选择的行。
    g.unselectRowByIndex(2);
cleanRow()
清除全部选中。
    g.cleanRow();
setParams(oParams)
  • oParams : Object 获取数据的参数对象 如 {"searchKeyWord":"test"};
设置dataSource的参数。
    var oparms = {"searchKeyWord":"test"};
    g.setParams(oparms);
toPage(pageNumber)
  • pageNumber : Number 页码
转到制定的页码。
    g.toPage(2);
setRowData(rowIndex,rowData)
  • rowIndex : Number 行序号
  • rowData : Object 行数据对象
设置指定行里面的内容。
   var rowData ={"sex":"1","age":"10","birthday":"2008-10-11","name":"test","status":"1"}
    g.setRowData(1,rowData);
setCellData(rowIndex,colId,v)
  • rowIndex : Number 行序号
  • colId : String 列ID
  • v : String cell里面的值
设置指定cell里面的内容。
    g.setCellData(1,'name','Jack');
getRowDomByIndex(rowIndex)
  • return : Object jQuery Element
  • rowIndex : Number 行序号
根据行号返回指定行jQuery Element对象。
    g.getRowDomByIndex(1);
getCellDomByIndex(rowIndex,colId)
  • return : Object jQuery Element
  • rowIndex : Number 行序号
  • colId : String 列id
根据行号和列的id得到一格的jQuery Element对象。
    g.getCellDomByIndex(1,'name');
getSelect()
  • return : Array 如[{"name":"jack"},{"name","mike"}]
获取当前选中行数据Array对象。
    g.getSelect();
toNextPage()
向下翻页。
    g.toNextPage();
toPrevPage()
向上翻页。
    g.toPrevPage();
toFirstPage()
回到第一页。
    g.toFirstPage();
toLastPage()
到最后页。
    g.toLastPage();
isFirstPage()
  • return : Boolean
是否在第一页。
    g.isFirstPage();
isLastPage()
  • return : Boolean
是否在最后页。
    g.isFirstPage();
getCurrentPage()
  • return : Number
得到当前的页码。
    g.getCurrentPage();
getTotalCount()
  • return : Number
得到总页码。
    g.getTotalCount();
getPageCount()
  • return : Number
得到当前页的数据数目。
    g.getPageCount();
